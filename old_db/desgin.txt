/*
    sales and sale_items table are related to each other.
    store data in sales and sale_items use procedure.

    --> generate_sale_id(g_pay_method payment_methods, g_phone TEXT, g_customer_id INT, v_sale_id OUT INT):

    this procedure for generate sale_id so customer can by many product with 1 sale_id
    --     g_pay_method: payment methods must between (cash, mobile_banking, card)
    --     g_phone     : give phone number if customer not have custmoer id else pass null,
    --     g_cust_id   : give customer id if customer have else pass null
    --     v_sale_id   : it's a output variable so you can extract sale_id


    --> record_items(sale_id INT, g_donut_id INT, g_quantity INT, v_total_price OUT DECIMAL)

    this function for record product it's and this procedure return total price of product
    --     g_sale_id  : sale_id from generate_sale_id,
    --     g_donut_id : which donut customer wants to buy give that id,
    --     g_quantity : how many quantity customer want to by
    --     v_total_price OUT DECIMAL: this is a output variable so you can extract total_price for each products

    this porcedure store many thinks automatic so you dont wary about.
    and store data in sales and sale_items tables.
*/

        example usage:
DO $$
DECLARE
    sale_id INT; -- declare for extract sale_id from procedure generate_sale_id()
    sub_total DECIMAL; -- declare for extract total price for each order items or product from procedure
    total_price DECIMAL DEFAULT 0; -- this is for calculating total price of all product
BEGIN
    CALL generate_sale_id('card', NULL, 1, sale_id); -- this procedure generating sale_id as i seyed before.

    CALL record_items(sale_id, 3, 5, sub_total); --
    total_price := total_price + sub_total;

    CALL record_items(sale_id, 2, 10, sub_total);
    total_price := total_price + sub_total;


    RAISE NOTICE 'sale id: %, total price of sale %', sale_id, total_price;
END;
$$;
